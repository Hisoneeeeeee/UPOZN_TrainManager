class Ticket:
    def init(self, departure_city, destination_city):
        self.departure_city = departure_city
        self.destination_city = destination_city

# Создание массива из структур билетов
tickets = [
    Ticket("Москва", "Санкт-Петербург"),
    Ticket("Санкт-Петербург", "Москва"),
    Ticket("Екатеринбург", "Новосибирск"),
    Ticket("Новосибирск", "Екатеринбург")
]

# Вывод доступных билетов
for ticket in tickets:
    print("Город отправления:", ticket.departure_city)
    print("Город назначения:", ticket.destination_city)
    print()

# Пример использования структур билетов
selected_ticket = tickets[0]
